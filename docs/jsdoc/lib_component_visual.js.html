     *  customer */
    cust: 3,

	 * usually loaded asset from gltf asset */
    mesh: 4,

    mesh_basic: 5,

    point: 6,

    refPoint: 7,

    cubeVoxel: 8,

    cubeVoxels: 9,
    /** html DOM as texture, test/example: htmltex.html */
    DomCanvas: 10,
    /** lines group, with dynamic lines' segment,

     */
    DynaSects: 11,
    /** opengl / webgl wireframe */
    wireframe: 12,
